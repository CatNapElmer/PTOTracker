/* ---------- Accrual computation ---------- */ function compute(){ let cumH=0, tSick=0, tPto=0; const rows = S.entries.map(e=>{ // Defensive calculations guarding against bad/mutated state parsing const hours = Math.max(0, parseFloat(e.hours) || 0); const sick = Math.max(0, parseFloat(e.sick) || 0); const pto = Math.max(0, parseFloat(e.pto) || 0); cumH += hours; tSick += sick; tPto += pto; // FIX: Calculate clean 30-hour blocks first, then multiply. const blocks = Math.floor(cumH / 30); const sickEarned = blocks * 1; const ptoEarned = blocks * 2.5; return { ...e, hours, sick, pto, cum: cumH.toFixed(2), sickBal: Math.max(0, sickEarned - tSick), ptoBal: Math.max(0, ptoEarned - tPto) }; }); const last = rows[rows.length-1] || null; return { rows, sickBal: last?last.sickBal:0, ptoBal: last?last.ptoBal:0, total: last?parseFloat(last.cum):0, }; }
